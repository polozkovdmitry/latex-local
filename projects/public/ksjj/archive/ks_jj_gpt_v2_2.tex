\section{Framework: Langevin dynamics}

Langevin dynamics define a class of second order SDEs characterized by the following (formal) stochastic Newton law:
\[
\gamma \ddot{X}_t^{\gamma} = F\bigl(t,X_t^{\gamma},\dot{X}_t^{\gamma}\bigr)\,dt + \sigma\bigl(t,X_t^{\gamma},\dot{X}_t^{\gamma}\bigr)\,\dot{L}_t,
\qquad t \ge 0.
\]

From a physical point of view, $X^{\gamma}=(X_t^{\gamma})_{t\ge0}$ models the position of a generic body evolving in $\mathbb{R}^d$,
$\dot{X}^{\gamma}=(\dot{X}_t^{\gamma})_{t\ge0}$ its velocity, $\gamma$ the volumic mass of the body,
$F:[0,\infty)\times\mathbb{R}^d\times\mathbb{R}^d\to\mathbb{R}^d$ an external force field
and $(L_t)_{t\ge0}$ a $m$-dimensional stochastic perturbation weighted by an inhomogeneous dispersion matrix~$\sigma$.
Setting $V^{\gamma}=\dot{X}^{\gamma}$, the above rewrites more rigorously into the $\mathbb{R}^{2d}$-valued SDE system:
\begin{equation}
\label{eq:1.1}
\begin{cases}
dX_t^{\gamma} = V_t^{\gamma}\,dt,\\
dV_t^{\gamma} = \gamma^{-1}F\bigl(t,X_t^{\gamma},V_t^{\gamma}\bigr)\,dt + \gamma^{-1}\sigma\bigl(t,X_t^{\gamma},V_t^{\gamma}\bigr)\,dL_t,
\end{cases}
\qquad t\ge0.
\end{equation}

In this formulation the coefficient $\gamma^{-1}$ is understood as a friction force.

\begin{remark}
A prototypical instance of \eqref{eq:1.1} is given by the following setting:

\begin{equation}
    \label{eq:1.2}
    \begin{aligned}
    &L = W \text{ is a $d$-dimensional Brownian motion}, \qquad
    F(t,x,v) = -\nabla f(x) - \lambda v, \\
    &\sigma(t,x,v) = \sigma I_{d\times d} > 0, \qquad \sigma \text{ is a scalar constant}.
    \end{aligned}
\end{equation}

In this situation \eqref{eq:1.1} writes as
\begin{equation}
\label{eq:1.3}
\begin{cases}
dX_t^{\gamma} = V_t^{\gamma}\,dt,\\
dV_t^{\gamma} = -\gamma^{-1}\bigl(\nabla f(X_t^{\gamma}) + V_t^{\gamma}\bigr)\,dt + \gamma^{-1}\sigma\,dW_t,
\end{cases}
\qquad t\ge0.
\end{equation}
\end{remark}

Langevin dynamics, originally introduced in~\cite{l08}, apply in modern science in a wide range of fields, including
fluid dynamics, biology, economy, social science~\cite{pt13,naldi10} and not so long ago, in artificial intelligence~\cite{ccb18}.

\section{Clarification on Pavliotis \texorpdfstring{\cite{p14}}{} scaling and limits}

(This section mainly refers to Chapter~6 in~\cite{p14}.) The Langevin model in~\cite{p14} (see Equation~(6.1), p.~181)
corresponds to a second order stochastic system describing the motion of a ``particle'' with
\[
d\dot{q}_t = -\nabla U(q_t)\,dt - \eta\dot{q}_t\,dt + \sqrt{\frac{2\eta}{\beta}}\,dW_t,
\qquad t\ge0,
\]
where $q=(q_t)_t$ is the position dynamics of the particle and $\dot{q}=(\dot{q}_t)_t$ the related velocity dynamics.
Motions are assumed to start at the position $q_0$ and velocity $\dot{q}_0$.
(For notation consistency, the notation of the potential force and the friction coefficient are here denoted by $\nabla U$ and $\eta$
instead of $\nabla V$ and $\gamma$ in~\cite{p14}.) To this model, we associate the family of rescalled dynamics
\[
q_t^{\eta}:=\lambda_{\eta}q_{t/\mu_{\eta}}, \qquad t\ge0.
\]

As such, the corresponding velocity $\dot{q}^{\eta}$ is given by
$\dot{q}_t^{\eta}=\dfrac{\lambda_{\eta}}{\mu_{\eta}}\dot{q}_{t/\mu_{\eta}}$ and so
\[
\begin{aligned}
\dot{q}_t^{\eta}
&= \frac{\lambda_{\eta}}{\mu_{\eta}}\dot{q}_0
- \frac{\eta\lambda_{\eta}}{\mu_{\eta}}\int_0^{t/\mu_{\eta}} \nabla U(q_s)\,ds
- \frac{\lambda_{\eta}}{\mu_{\eta}}\int_0^{t/\mu_{\eta}} \dot{q}_s\,ds
+ \frac{\lambda_{\eta}}{\mu_{\eta}}\sqrt{\frac{2\eta}{\beta}}\,W_{t/\mu_{\eta}} \\
&= \frac{\lambda_{\eta}}{\mu_{\eta}}\dot{q}_0
- \frac{\lambda_{\eta}}{(\mu_{\eta})^2}\int_0^{t} \underbrace{\nabla U(q_{s/\mu_{\eta}})}_{=\nabla U(q_s^{\eta}/\lambda_{\eta})}\,ds
- \frac{\eta\lambda_{\eta}}{(\mu_{\eta})^2}\int_0^{t} \underbrace{\dot{q}_{s/\mu_{\eta}}}_{=(\mu_{\eta}/\lambda_{\eta})\dot{q}_s^{\eta}}\,ds
+ \frac{\lambda_{\eta}}{\mu_{\eta}}\sqrt{\frac{2\eta}{\beta}}\,W_{t/\mu_{\eta}}.
\end{aligned}
\]

Since (see e.g.~\cite{ks91}) $(W_{t/\mu_{\eta}})_{t\ge0}\overset{\mathcal{D}}{=}(W_t/\sqrt{\mu_{\eta}})_{t\ge0}$, $\dot q^{\eta}$ can be viewed as solution to the SDE:
\[
d\dot q_t^{\eta}
=
-\frac{\lambda_{\eta}}{(\mu_{\eta})^2}\nabla U\!\left(q_t^{\eta}/\lambda_{\eta}\right)dt
-\frac{\eta}{\mu_{\eta}}\dot q_t^{\eta}\,dt
+\frac{\lambda_{\eta}}{(\mu_{\eta})^{3/2}}\sqrt{\frac{2\eta}{\beta}}\,dW_t,
\qquad t\ge0,
\qquad
\dot q_0^{\eta}=\frac{\lambda_{\eta}}{\mu_{\eta}}\dot q_0.
\]

In particular, taking $\lambda_{\eta}=1$ and $\mu_{\eta}=1/\eta$, the above writes as
\[
d\dot q_t^{\eta}
=
-\eta^2\nabla U(q_t^{\eta})\,dt-\eta^2\dot q_t^{\eta}\,dt+\eta^2\sqrt{\frac{2}{\beta}}\,dW_t,
\qquad t\ge0,
\qquad
\dot q_0^{\eta}=\eta\dot q_0.
\]

Comparing the above with \eqref{eq:1.1} (and as intuited above), the (friction) coefficient $\eta^2$ corresponds to the inverse of the mass $\gamma$.
(While the rescaling in~\cite{p14} are seemingly coherent with our initial Langevin framework, the remaining impact on the initial velocity seems to have been overlooked: the Kramers-Smoluchowski limit $\gamma\downarrow 0$ or equivalently $\eta\uparrow\infty$ makes $\dot q_0^{\eta}=\eta\dot q_0$ blows up which greatly contrasts with Proposition~\ref{prop:ks} below \dots\ The initial velocity $\dot q_0$ should have been redefined as $\dot q_0=V_0/\eta$ for some fixed random $V_0\in\mathbb{R}^d$.)

\section{Quantitative Kramers-Smoluchowski limit}

In the proposition below, under the prototypical condition \eqref{eq:1.2}, we establish the so-called Kramers-Smoluchowski limit asserting the $\gamma\downarrow 0$ (zero-mass) limit of \eqref{eq:1.1}.

\begin{proposition}\label{prop:ks}
Let $(X^{\gamma},V^{\gamma})$ be the Langevin dynamic \eqref{eq:1.3}.
Assume that the initial condition $(X_0^{\gamma},V_0^{\gamma})$ are independent of $\gamma$, set at $(X_0,V_0)$ with $\mathbb{E}[|X_0|^2+|V_0|^2]<\infty$, and that $\sigma\neq 0$ and $\nabla f$ is Lipschitz-continuous.
Then for all finite time-horizon $T$, $\gamma\le 1$ and integer $p\ge1$,
\begin{equation}
\label{eq:3.4}
\max_{t\in[0,T]}\mathbb{E}\bigl[|X_t^{\gamma}-X_t^0|^{2p}\bigr]=\mathcal{O}(\gamma^p),
\end{equation}
where $X^0$ is the solution to
\begin{equation}
\label{eq:3.5}
dX_t^0=-\nabla f(X_t^0)\,dt+\sigma\,dW_t,\qquad X_0^0=X_0.
\end{equation}
Moreover,
\begin{equation}
\label{eq:3.6}
\mathbb{E}\Bigl[\max_{t\in[0,T]}|X_t^{\gamma}-X_t^0|^2\Bigr]=\mathcal{O}(\gamma^{1/2}).
\end{equation}
\end{proposition}

While the Kramers-Smoluchowski limit is more directly linked to \eqref{eq:3.6}, the proposition is purposely written to emphasize the successive ``easy'' part \eqref{eq:3.4}\footnote{Stated without proof details in \cite{p14}, bottom of p.~206.} and more ``tricky'' part \eqref{eq:3.6} of the Kramers-Smoluchowski limit.
In particular, for $p=1$, \eqref{eq:3.4} is optimal but the discrepancy between the rate $\mathcal{O}(\gamma)$ in \eqref{eq:3.4} and the $\mathcal{O}(\gamma^{1/2})$ in \eqref{eq:3.6}, although reflecting the cost between time marginal and max-in-time mean-square distance (which may be significant), is rather non-intuitive.
The demonstration of \eqref{eq:3.6} might be so subject to some proof refinements.

The discrepancy between the rate $\mathcal{O}(\gamma)$ in \eqref{eq:3.4} and the $\mathcal{O}(\gamma^{1/2})$ in \eqref{eq:3.6} is rather non-intuitive (and is possibly subject from some refinements) but reflect the cost between time marginal and max-in-time mean-square distance.
\end{remark}

\paragraph{Proof.}
As a preliminary, recall that, under the initial and Lipschitz conditions of the proposition, each Equations \eqref{eq:1.1} and \eqref{eq:3.5} admit a unique solution with finite second moments at all times.
In particular, as $\nabla f$ is Lipschitz, $\nabla f$ grows (at most) linearly with
\begin{equation}
\label{eq:3.7}
|\nabla f(x)| \le |\nabla f(0)| + |\nabla f|_{\mathrm{Lip}}|x|
\le \max\bigl(|\nabla f(0)|,|\nabla f|_{\mathrm{Lip}}\bigr)\bigl(1+|x|\bigr),
\end{equation}
for $|\nabla f|_{\mathrm{Lip}}=\sup_{x\neq y}\dfrac{|\nabla f(x)-\nabla f(y)|}{|x-y|}$.
Consequently (after some calculations on the SDE \eqref{eq:3.5}, see e.g.~\cite{ks91}), we get, for all integer $p\ge1$,
\begin{equation}
\label{eq:3.8}
\max_{t\in[0,T]}\mathbb{E}\bigl[|X_t^0|^{2p}\bigr]<\infty.
\end{equation}

{\color{teal}
\noindent\emph{Comment.} The only input here is the standard polynomial-moment estimate for an It\^o diffusion with globally Lipschitz drift and constant diffusion. Indeed, by \eqref{eq:3.7}, the drift in \eqref{eq:3.5} has linear growth, so It\^o's formula applied to $|X_t^0|^{2p}$ gives a differential inequality of the form
\[
\frac{d}{dt}\mathbb{E}|X_t^0|^{2p}\le C_p\bigl(1+\mathbb{E}|X_t^0|^{2p}\bigr),
\]
and Gronwall yields \eqref{eq:3.8}. No special structure of $f$ is used beyond global Lipschitz continuity of $\nabla f$.
}

\underline{Proof of \eqref{eq:3.4}:}
The limit \eqref{eq:3.4} is essentially established by stochastic calculus. Define $U_t=\exp\{\gamma^{-1}t\}V_t^{\gamma}$
and applying It\^o's formula immediately observe that, by It\^o's integration by parts (see Appendix)
\[
dU_t=-\gamma^{-1}\exp\{\gamma^{-1}t\}\nabla f(X_t^{\gamma})\,dt+\sigma\gamma^{-1}\exp\{\gamma^{-1}t\}\,dW_t,
\]
or equivalently
\[
U_t=V_0+\gamma^{-1}\int_0^t \exp\{\gamma^{-1}s\}\nabla f(X_s^{\gamma})\,ds+\sigma\gamma^{-1}\int_0^t \exp\{\gamma^{-1}s\}\,dW_s.
\]

{\color{teal}
\noindent\emph{Comment.} This is the usual variation-of-constants step for the linear velocity equation. One multiplies the $V^\gamma$-equation by the integrating factor $e^{t/\gamma}$ so that the friction term is cancelled. The resulting identity is the explicit Duhamel formula for $V_t^\gamma$.
}

It thus follows that
\begin{equation}
\label{eq:3.9}
\begin{aligned}
V_t^{\gamma} &= \exp\{-\gamma^{-1}t\}U_t \\
&= \exp\{-\gamma^{-1}t\}V_0+\gamma^{-1}\int_0^t \exp\{\gamma^{-1}(s-t)\}\nabla f(X_s^{\gamma})\,ds+\sigma\gamma^{-1}\int_0^t \exp\{\gamma^{-1}(s-t)\}\,dW_s,
\end{aligned}
\end{equation}
and, further,
\[
\begin{aligned}
X_t^{\gamma} &= X_0^{\gamma}+\int_0^t V_s^{\gamma}\,ds \\
&= X_0^{\gamma}+\int_0^t \exp\{-\gamma^{-1}s\}\,ds\,V_0 \\
&\quad -\gamma^{-1}\int_0^t \left(\int_s^t \exp\{\gamma^{-1}(s-r)\}\,dr\right)\nabla f(X_s^{\gamma})\,ds+\sigma\gamma^{-1}\int_0^t \left(\int_s^t \exp\{\gamma^{-1}(s-r)\}\,dr\right)\,dW_s \\
&= X_0^{\gamma}+\gamma V_0\bigl(1-\exp\{-\gamma^{-1}t\}\bigr) \\
&\quad +\int_0^t \bigl(1-\exp\{\gamma^{-1}(s-t)\}\bigr)\nabla f(X_s^{\gamma})\,ds+\sigma\int_0^t \bigl(1-\exp\{\gamma^{-1}(s-t)\}\bigr)\,dW_s.
\end{aligned}
\]

{\color{teal}
\noindent\emph{Comment.} To pass from \eqref{eq:3.9} to the representation of $X_t^\gamma$, integrate in time and exchange the order of integration in the deterministic and stochastic double integrals. For the deterministic term this is the usual Fubini theorem on the triangle $\{0\le s\le r\le t\}$; for the stochastic term one uses stochastic Fubini, which is legitimate here because the kernel $e^{(s-r)/\gamma}\mathbf 1_{\{s\le r\le t\}}$ is square-integrable. The elementary kernel identity
\[
\gamma^{-1}\int_s^t e^{(s-r)/\gamma}\,dr
=1-e^{(s-t)/\gamma}
\]
then produces the factor $1-e^{(s-t)/\gamma}$.
}

Comparing the paths of $X$ and $X^0$, we get
\begin{equation}
\label{eq:3.10}
\begin{aligned}
X_t^{\gamma}-X_t^0
&= \gamma V_0\bigl(1-\exp\{-\gamma^{-1}t\}\bigr)+\int_0^t \bigl(\nabla f(X_s^{\gamma})-\nabla f(X_s^0)\bigr)\,ds \\
&\quad -\int_0^t \exp\{\gamma^{-1}(s-t)\}\nabla f(X_s^{\gamma})\,ds-\sigma\int_0^t \exp\{\gamma^{-1}(s-t)\}\,dW_s,
\end{aligned}
\end{equation}
and, equivalently,
\[
X_t^{\gamma}-X_t^0=\int_0^t \bigl(\nabla f(X_s^{\gamma})-\nabla f(X_s^0)\bigr)\,ds+\gamma\bigl(V_0-V_t^{\gamma}\bigr).
\]

{\color{teal}
\noindent\emph{Comment.} The first line is obtained by subtracting the mild form of $X^0$,
\[
X_t^0=X_0-\int_0^t \nabla f(X_s^0)\,ds+\sigma W_t,
\]
from the previous representation of $X_t^\gamma$, and regrouping the terms involving $\nabla f(X_s^\gamma)$ and $\nabla f(X_s^0)$. The ``equivalently'' identity is simply a re-expression of \eqref{eq:3.9}: multiplying \eqref{eq:3.9} by $\gamma$ and rearranging gives
\[
\gamma(V_0-V_t^\gamma)
=
\gamma V_0\bigl(1-e^{-t/\gamma}\bigr)
-\int_0^t e^{(s-t)/\gamma}\nabla f(X_s^\gamma)\,ds
-\sigma\int_0^t e^{(s-t)/\gamma}\,dW_s,
\]
which is exactly the remainder term in \eqref{eq:3.10}. So the difference $X_t^\gamma-X_t^0$ splits into a Lipschitz drift error plus a fast-velocity boundary layer term.
}

Taking the Euclidean norm $|\cdot|$, applying the triangular inequality, elevating the resulting expression to the power $2p$ and taking the expectation yields
\begin{equation}
\label{eq:3.11}
\mathbb{E}\bigl[|X_t^{\gamma}-X_t^0|^{2p}\bigr]\le 2^{2p-1}\left(\mathbb{E}\left[\left|\int_0^t \bigl(\nabla f(X_s^{\gamma})-\nabla f(X_s^0)\bigr)\,ds\right|^{2p}+\bigl(\gamma|V_0-V_t^{\gamma}|\bigr)^{2p}\right]\right).
\end{equation}

{\color{teal}
\noindent\emph{Comment.} The inequality used is the standard convexity estimate
\[
|a+b|^{2p}\le 2^{2p-1}\bigl(|a|^{2p}+|b|^{2p}\bigr).
\]
No martingale structure enters here yet.
}

Using H\"older's inequality and the Lipschitz condition of $\nabla f$,
\[
\mathbb{E}\left[\left|\int_0^t \bigl(\nabla f(X_s^{\gamma})-\nabla f(X_s^0)\bigr)\,ds\right|^{2p}\right]\le |\nabla f|_{\mathrm{Lip}}^{2p} t^{2p-1}\int_0^t |X_s^{\gamma}-X_s^0|^{2p}\,ds.
\]

{\color{teal}
\noindent\emph{Comment.} This is the deterministic estimate
\[
\Bigl|\int_0^t h_s\,ds\Bigr|^{2p}\le t^{2p-1}\int_0^t |h_s|^{2p}\,ds,
\]
which is just H\"older/Jensen on the measure $ds/t$, followed by $| \nabla f(x)-\nabla f(y)|\le |\nabla f|_{\mathrm{Lip}}|x-y|$.
}

Meanwhile, recalling \eqref{eq:3.9}, we have
\[
\begin{aligned}
\mathbb{E}\bigl[\gamma^{2p}|V_0-V_t^{\gamma}|^{2p}\bigr]
&\le 3^{2p-1}\Bigl(\gamma^{2p}\bigl(1-\exp\{-\gamma^{-1}t\}\bigr)^{2p}\mathbb{E}[|V_0|^{2p}] \\
&\qquad\qquad +\mathbb{E}\left[\left|\int_0^t \exp\{\gamma^{-1}(s-t)\}\nabla f(X_s^{\gamma})\,ds\right|^{2p}\right] \\
&\qquad\qquad +\sigma^{2p}\mathbb{E}\left[\left|\int_0^t \exp\{\gamma^{-1}(s-t)\}\,dW_s\right|^{2p}\right]\Bigr).
\end{aligned}
\]

{\color{teal}
\noindent\emph{Comment.} This is again the elementary inequality
\[
|a+b+c|^{2p}\le 3^{2p-1}\bigl(|a|^{2p}+|b|^{2p}+|c|^{2p}\bigr),
\]
applied to the three pieces in the explicit representation of $\gamma(V_0-V_t^\gamma)$.
}

Adding and subtracting $\nabla f(X_s^0)$, using the Lipschitz property of $\nabla f$ and the Jensen inequality, and observing that $\exp\{\gamma^{-1}(s-t)\}\le 1$ for all $s\le t$ yield
\[
\begin{aligned}
\mathbb{E}\left[\left|\int_0^t \exp\{\gamma^{-1}(s-t)\}\nabla f(X_s^{\gamma})\,ds\right|^{2p}\right]
&=\mathbb{E}\Biggl[\Biggl|\int_0^t \exp\{\gamma^{-1}(s-t)\}\bigl(\nabla f(X_s^{\gamma})-\nabla f(X_s^0)\bigr)\,ds \\
&\qquad\qquad +\int_0^t \exp\{\gamma^{-1}(s-t)\}\nabla f(X_s^0)\,ds\Biggr|^{2p}\Biggr] \\
&\le 2^{2p-1}|\nabla f|_{\mathrm{Lip}}^{2p}\mathbb{E}\left[\int_0^t |X_s^{\gamma}-X_s^0|^{2p}\,ds\right] \\
&\qquad +2^{2p-1}\mathbb{E}\left[\int_0^t \exp\{2p\gamma^{-1}(s-t)\}|\nabla f(X_s^0)|^{2p}\,ds\right].
\end{aligned}
\]

{\color{teal}
\noindent\emph{Comment.} The first term is controlled exactly as before, now with the extra kernel bounded by $1$. For the second term, the point is that Jensen on the probability measure with density proportional to $e^{\gamma^{-1}(s-t)}\,ds$ converts the $2p$-th power of the time integral into the time integral of the $2p$-th power, at the price of changing the kernel to $e^{2p\gamma^{-1}(s-t)}$. Any harmless multiplicative factor depending on $T$ can be absorbed into the generic constant $C$ later.
}

Furthermore, according to the property \eqref{eq:3.7}, we have
\[
\begin{aligned}
\mathbb{E}\left[\int_0^t \exp\{2p\gamma^{-1}(s-t)\}|\nabla f(X_s^0)|^{2p}\,ds\right]
&\le \frac{\gamma}{2p}\max\bigl(|\nabla f(0)|^{2p},|\nabla f|_{\mathrm{Lip}}^{2p}\bigr)\bigl(1-\exp\{-2p\gamma^{-1}t\}\bigr) \\
&\qquad\times \mathbb{E}\left[1+\max_{0\le s\le t}|X_s^0|^{2p}\right].
\end{aligned}
\]

{\color{teal}
\noindent\emph{Comment.} Here one simply inserts the linear-growth bound \eqref{eq:3.7} in the form
\[
|\nabla f(X_s^0)|^{2p}\le C\bigl(1+|X_s^0|^{2p}\bigr)\le C\Bigl(1+\max_{0\le r\le t}|X_r^0|^{2p}\Bigr),
\]
pulls the random factor outside the $s$-integral, and computes the remaining deterministic integral explicitly:
\[
\int_0^t e^{2p(s-t)/\gamma}\,ds=\frac{\gamma}{2p}\bigl(1-e^{-2pt/\gamma}\bigr).
\]
So this whole contribution is of order $\gamma$.
}

Finally, using Burkh\"older-Davis-Gundy inequality, we have
\[
\mathbb{E}\left[\left|\int_0^t \exp\{\gamma^{-1}(s-t)\}\,dW_s\right|^{2p}\right]
\le (2dp)^p\mathbb{E}\left[\left(\int_0^t \exp\{2\gamma^{-1}(s-t)\}\,ds\right)^p\right]
=2^{p-1}(dp)^p\gamma^p\bigl(1-\exp\{-2\gamma^{-1}t\}\bigr)^p.
\]

{\color{teal}
\noindent\emph{Comment.} This is the standard BDG estimate for a stochastic integral with deterministic integrand. Since the quadratic variation equals
\[
\int_0^t e^{2(s-t)/\gamma}\,ds=\frac{\gamma}{2}\bigl(1-e^{-2t/\gamma}\bigr),
\]
the $2p$-th moment is of order $\gamma^p$. This is the first place where the fast relaxation scale $\gamma$ appears quantitatively.
}

Combining all the above, we first deduce that
\[
\mathbb{E}\bigl[\gamma^{2p}|V_0-V_t^{\gamma}|^{2p}\bigr]\le C\left(\gamma^{2p}\mathbb{E}[|V_0|^{2p}]+\int_0^t |X_s^{\gamma}-X_s^0|^{2p}\,ds+\gamma\left(\mathbb{E}\left[1+\max_{0\le s\le t}|X_s^0|^{2p}\right]+\sigma^{2p}\right)\right)
\]
for some constant $C$ depending only on $p,d$ and $\nabla f$, and coming back to \eqref{eq:3.11} - and up to a slight change of the factor $C$, we obtain that, for any $\gamma\le 1$,
\begin{equation}
\label{eq:3.12}
\mathbb{E}\bigl[|X_t^{\gamma}-X_t^0|^{2p}\bigr]\le C\gamma^p+\int_0^t \mathbb{E}\bigl[|X_s^{\gamma}-X_s^0|^{2p}\bigr]ds.
\end{equation}

{\color{teal}
\noindent\emph{Comment.} The bookkeeping up to this point is straightforward except for one point of rate. By \eqref{eq:3.8}, the moment
\[
\mathbb{E}\Bigl[1+\max_{0\le s\le t}|X_s^0|^{2p}\Bigr]
\]
is bounded uniformly on $[0,T]$, so the last parenthesis is harmless. However, the displayed bounds as written naturally produce a remainder of order $O(\gamma)+O(\gamma^p)$, because the deterministic convolution against $\nabla f(X_s^0)$ was only estimated using the mass of the exponential kernel. Therefore, if one wants the sharp statement \eqref{eq:3.12} with $C\gamma^p$ for every $p\ge1$, one needs at this stage a slightly sharper estimate of that deterministic term than the one displayed above. The rest of the argument is then exactly a Gronwall argument. I am flagging this because it is the only place in the proof where one should be a little careful not to lose powers of $\gamma$.
}

Using Gronwall's inequality, \eqref{eq:3.4} follows.

\underline{Proof of \eqref{eq:3.6}:}
Before establishing \eqref{eq:3.6}, let us point out that, contrary to some papers (e.g.~\cite{f04}), the limit is so directly deduced from martingale inequalities and needs a few careful steps.
Coming back to \eqref{eq:3.10}, we deduce that, for all intermediate time-horizon $T_0\in(0,T]$
\[
\begin{aligned}
\max_{t\in[0,T_0]}|X_t^{\gamma}-X_t^0|^2
&\le \gamma^2\max_{t\in[0,T_0]}\bigl(1-\exp\{-\gamma^{-1}t\}\bigr)^2|V_0|^2+2\max_{t\in[0,T_0]}\int_0^t |X_s^{\gamma}-X_s^0|^2\,ds \\
&\quad -2\sigma\min_{t\in[0,T_0]}\int_0^t \exp\{\gamma^{-1}(s-t)\}\bigl(X_s^{\gamma}-X_s^0\bigr)\cdot dW_s \\
&\quad +2\max_{t\in[0,T_0]}\int_0^t \exp\{\gamma^{-1}(s-t)\}\max\bigl(|\nabla f(0)|^2,|\nabla f|_{\mathrm{Lip}}^2\bigr)(1+|X_s^0|)^2\,ds \\
&\quad +\max_{t\in[0,T_0]}\frac{\gamma\sigma^2}{2}\bigl(1-\exp\{-2\gamma^{-1}t\}\bigr) \\
&\le \beta_{T_0}^{\gamma}+2\int_0^{T_0}|X_s^{\gamma}-X_s^0|^2\,ds-2\sigma\min_{t\in[0,T_0]}\int_0^t \exp\{\gamma^{-1}(s-t)\}\bigl(X_s^{\gamma}-X_s^0\bigr)\cdot dW_s,
\end{aligned}
\]
for
\[
\beta_t^{\gamma}:=\gamma^2|V_0|^2+4\gamma\max\bigl(|\nabla f(0)|^2,|\nabla f|_{\mathrm{Lip}}^2\bigr)\int_0^t (1+|X_s^0|)^2\,ds+\frac{\gamma\sigma^2}{2}.
\]

Taking the expectation of the last inequality gives
\[
\mathbb{E}\left[\max_{t\in[0,T_0]}|X_t^{\gamma}-X_t^0|^2\right]\le \mathbb{E}[\beta_{T_0}^{\gamma}]+2\int_0^{T_0}\mathbb{E}\bigl[|X_s^{\gamma}-X_s^0|^2\bigr]ds-2\sigma\mathbb{E}\left[\min_{t\in[0,T_0]}\int_0^t \exp\{\gamma^{-1}(s-t)\}\bigl(X_s^{\gamma}-X_s^0\bigr)\cdot dW_s\right].
\]

At this stage, a simple upper-bound of order $\gamma$ can be derived for $\mathbb{E}[\beta_{T_0}^{\gamma}]$ (see below) and the main technical lies with the handling of the remaining ``stochastic integral''\footnote{The exact terminology should rather be stochastic convolution.}
\[
I_t^{\gamma}:=\int_0^t \exp\{\gamma^{-1}(s-t)\}\bigl(X_s^{\gamma}-X_s^0\bigr)\cdot dW_s,\qquad t\ge 0.
\]

Introducing the process $\bigl(L_t^{\gamma}:=\int_0^t (X_s^{\gamma}-X_s^0)\cdot dW_s\bigr)_{t\ge0}$, $I^{\gamma}$ writes as
\[
I_t^{\gamma}=\exp\{-\gamma^{-1}t\}L_t^{\gamma},\qquad t\ge0,
\]
and it can be observed (by It\^o's integration by parts) that $I^{\gamma}$ is also solution to the SDE:
\[
dI_t^{\gamma}=-\gamma^{-1}I_t^{\gamma}\,dt+dL_t^{\gamma},\qquad t\ge0,\qquad I_0=0.
\]

By It\^o's formula, this specifically yields
\[
d|I_t^{\gamma}|^2=2I_t^{\gamma}\,dI_t^{\gamma}+d\langle I^{\gamma}\rangle_t
=\bigl(|X_t^{\gamma}-X_t^0|^2-2\gamma^{-1}|I_t^{\gamma}|^2\bigr)\,dt+2I_t^{\gamma}\,dL_t^{\gamma},
\]
and further
\[
|I_t^{\gamma}|^2+2\gamma^{-1}\int_0^t |I_s^{\gamma}|^2\,ds
=\int_0^t |X_s^{\gamma}-X_s^0|^2\,ds+2\int_0^t I_s^{\gamma}\,dL_s^{\gamma}.
\]

Therefore, for all $T_0\le T$,
\[
\mathbb{E}\left[\max_{t\in[0,T_0]}|I_t^{\gamma}|^2\right]
\le \int_0^{T_0}\mathbb{E}\bigl[|X_t^{\gamma}-X_t^0|^2\bigr]dt
+2\mathbb{E}\left[\max_{t\in[0,T_0]}\int_0^t I_s^{\gamma}\,dL_s^{\gamma}\right]
\]
where, applying successively the Cauchy-Schwarz and Burkh\"older-Davis-Gundy inequality,
\begin{equation}
\label{eq:3.13}
\mathbb{E}\left[\max_{t\in[0,T_0]}\int_0^t I_s^{\gamma}\,dL_s^{\gamma}\right]
\le \left(\mathbb{E}\left[\max_{t\in[0,T_0]}\left(\int_0^t I_s^{\gamma}\,dL_s^{\gamma}\right)^2\right]\right)^{1/2}
\le 2\left(\mathbb{E}\left[\int_0^{T_0}|I_s^{\gamma}|^2|X_s^{\gamma}-X_s^0|^2\,ds\right]\right)^{1/2}.
\end{equation}

According to Jensen's inequality, $\mathbb{E}[\max_{t\in[0,T_0]}|I_t^{\gamma}|^2]\ge \bigl(\mathbb{E}[\max_{t\in[0,T_0]}|I_t^{\gamma}|]\bigr)^2$ and so
\[
\begin{aligned}
\left(\mathbb{E}\left[\max_{t\in[0,T_0]}|I_t^{\gamma}|\right]\right)^2
&\le \int_0^{T_0}\mathbb{E}\bigl[|X_t^{\gamma}-X_t^0|^2\bigr]dt+2\left(\mathbb{E}\left[\int_0^{T_0}|I_s^{\gamma}|^2|X_s^{\gamma}-X_s^0|^2\,ds\right]\right)^{1/2} \\
&\le \int_0^{T_0}\mathbb{E}\bigl[|X_t^{\gamma}-X_t^0|^2\bigr]dt \\
&\quad +\sqrt{2}\left(\int_0^{T_0}\mathbb{E}\left[\max_{v\in[0,s]}|I_v^{\gamma}|^4\right]ds+\int_0^{T_0}\mathbb{E}\bigl[|X_s^{\gamma}-X_s^0|^4\bigr]ds\right)^{1/2},
\end{aligned}
\]
using the inequality: for all $a,b\in\mathbb{R}$, $ab\le \tfrac12(a^2+b^2)$.
Taking the square of the previous estimate, we deduce that
$f(t):=\bigl(\mathbb{E}[\max_{s\in[0,t]}|I_s^{\gamma}|]\bigr)^4$ satisfies an inequality of the form: for all $0\le T_0\le T$,
\[
f(T_0)\le J_{\gamma}(T_0)+4\int_0^{T_0}f(s)\,ds
\]
for
\[
J_{\gamma}(t)=4\left(\left(\int_0^{T_0}\mathbb{E}\bigl[|X_s^{\gamma}-X_s^0|^2\bigr]ds\right)^2+\int_0^{T_0}\mathbb{E}\bigl[|X_s^{\gamma}-X_s^0|^4\bigr]ds\right).
\]
Using Gronwall's inequality, it follows that $f(T_0)\le J_{\gamma}(T_0)\exp\{4T_0\}$ and further
\[
\mathbb{E}\left[\max_{t\in[0,T_0]}|I_t^{\gamma}|\right]
\le \bigl(J_{\gamma}(T_0)\bigr)^{1/4}\exp\{T\}
\le \sqrt{2}\left(\int_0^{T_0}\mathbb{E}\bigl[|X_s^{\gamma}-X_s^0|^4\bigr]ds\right)^{1/4}\exp\{T_0\}.
\]

Coming back to the initial estimate on $\mathbb{E}[\max_{t\in[0,T_0]}|X_t^{\gamma}-X_t^0|^2]$, we get
\[
\mathbb{E}\left[\max_{t\in[0,T_0]}|X_t^{\gamma}-X_t^0|^2\right]
\le \mathbb{E}[\beta_{T_0}^{\gamma}]
+2\int_0^{T_0}\mathbb{E}\bigl[|X_s^{\gamma}-X_s^0|^2\bigr]ds
+2\sqrt{2}\sigma\left(\int_0^{T_0}\mathbb{E}\bigl[|X_s^{\gamma}-X_s^0|^4\bigr]ds\right)^{1/4}\exp\{T_0\}.
\]

Given $\gamma\le1$, $\mathbb{E}[\beta_{\gamma}(t)]\le C\gamma$ and according to \eqref{eq:3.4},
\[
\int_0^{T_0}\mathbb{E}\bigl[|X_s^{\gamma}-X_s^0|^2\bigr]ds\le C\gamma,
\qquad
\left(\int_0^{T_0}\mathbb{E}\bigl[|X_s^{\gamma}-X_s^0|^4\bigr]ds\right)^{1/4}\le C\gamma^{1/2}.
\]

With this, the rate $\mathcal{O}(\gamma^{1/2})$ leading to \eqref{eq:3.6} follows. \hfill $\square$

\noindent\textbf{Gronwall's lemma:}
Let $f,\alpha,g$ be three scalar functions, all defined on a certain interval $[0,T]$ and which are such that
\[
\forall\, t\in[0,T], \qquad f(t)\le g(t)+\int_0^t \alpha(s)f(s)\,ds.
\]
Then if $g$ is non-decreasing and $\alpha\ge0$, we have
\[
\forall\, t\in[0,T], \qquad f(t)\le g(t)\exp\left\{\int_0^t \alpha(s)\,ds\right\}.
\]

\noindent\textbf{Covariations/Crossvariation process and quadratic variation:}
Assume that $X$ and $Y$ are two real-valued continuous semi-martingales given by:
\[
X_t=X_0+M_t^X+A_t^X,\qquad Y_t=X_0+M_t^Y+A_t^Y,\qquad t\ge0,
\]
with $M^X$ and $M^Y$ being the respective martingale parts of $X$ and $Y$ and $A^X$ and $A^Y$ the finite variation parts.
Then the covariation (crossvariation) process of $X$ and $Y$ is the stochastic process $\langle X,Y\rangle=(\langle X,Y\rangle_t)_{t\ge0}$ defined at all time $t$ by
\[
\langle X,Y\rangle_t=\limsup_{|t_{i+1}-t_i|\to0}\sum_{k=0}^{N-1}(X_{t_{k+1}}-X_{t_k})(Y_{t_{k+1}}-Y_{t_k}).
\]

The particular process $\langle X\rangle=\langle X,X\rangle$ corresponds to the quadratic variation process of $X$.

\noindent\textbf{It\^o's integration by parts:}
Let $(X_t;\,t\ge0)$ and $(Y_t;\,t\ge0)$ be two continuous semi-martingales. Then a.s., for all $t\ge0$,
\[
X_tY_t=X_0Y_0+\int_0^t X_s\,dY_s+\int_0^t Y_s\,dX_s+\langle X,Y\rangle_t.
\]

In the particular case where $X_t=\int_0^t \sigma_s\,dW_s$ and $Y_t=\int_0^t b_s\,ds$ with $\sigma$ and $b$ integrable $\mathcal{F}_t$-adapted processes, the above gives:
\[
X_tY_t=\int_0^t X_sb_s\,ds+\int_0^t Y_s\sigma_s\,dW_s,\qquad t\ge0.
\]

\noindent\textbf{It\^o's formula for general semi-martingale processes:}
Let $(X_t;\,t\ge0)$ be a $\mathbb{R}^d$-valued continuous semi-martingale with the decomposition:
\[
X_t=X_0+M_t+A_t,\qquad t\ge0,
\]
where $M$ is a martingale and $A$ a c\`adl\`ag process with finite variations. Then, for all scalar function
$f:[0,T]\times\mathbb{R}^d\to\mathbb{R}$, satisfying the following properties:
\begin{itemize}
\item for all $x$, $t\mapsto f(t,x)$ is differentiable and the partial derivative $(t,x)\mapsto \partial_t f(t,x)$ is continuous,
\item for all $t$, $x\mapsto f(t,x)$ is twice continuously differentiable on $\mathbb{R}^d$ and, for all $1\le i,j\le d$, $(t,x)\mapsto \partial_{x_i x_j}^2 f(t,x)$ is continuous,
\end{itemize}
it holds
\[
f(t,X_t)=f(0,X_0)+\int_0^t \partial_s f(s,X_s)\,ds+\sum_i\int_0^t \partial_{x_i}f(s,X_s)\,dX_s^i+\frac{1}{2}\sum_{i,j}\int_0^t \partial_{x_i x_j}^2 f(s,X_s)\,d\langle X^i,X^j\rangle_s.
\]

\noindent\textbf{Equivalent formulation:}
For $\nabla f$ the gradient of $f$, $\nabla_x^2f$ the Hessian of $f$ and $\langle X,X\rangle_t$ the $d\times d$ covariation matrix of $X$ (i.e. $\langle X,X\rangle_t^{(i,j)}=\langle X^i,X^j\rangle_t$)
\[
\begin{aligned}
f(t,X_t)
&=f(0,X_0)+\int_0^t \partial_s f(s,X_s)\,ds+\int_0^t \nabla_x f(s,X_s)\,dM_s \\
&\quad +\int_0^t \nabla_x f(s,X_s)\,dA_s+\frac{1}{2}\int_0^t \mathrm{Trace}\bigl(\nabla_x^2 f(s,X_s)\,d\langle X,X\rangle_s\bigr).
\end{aligned}
\]
(this formulation particularly yields: under some general assumptions, the transformation $f(t,X_t)$ of a semi-martingale remains a semi-martingale.)
